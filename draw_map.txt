draw_map:
   mov si, 0
.new_row:
   mov di, 0
.next_col:
   mov ax, si
   mov bx, 10
   mul bx
   mov bx, 320
   mul bx
   mov dx, ax

   mov ax, di
   mov bx, 10
   mul bx

   add ax, dx

   mov bx, DRAW_START
   add bx, ax
   mov byte [bx], COLOR_WHITE

   inc di
   cmp di, 32
   jl .next_col

   inc si
   cmp si, 20
   jl .new_row

   jmp done
